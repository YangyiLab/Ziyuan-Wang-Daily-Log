$$a^{(t)} = b + \textbf{W} h^{(t-1)} + \textbf{U} x^{(t)}\\

  o^{(t)} = c + \textbf{V}h^{(t)}, \\
  \hat y^{(t)} = {softmax}(o^{(t)}) $$
